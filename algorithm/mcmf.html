#include <iostream>
#include <vector>
#include <queue>
#include <cstring>

using namespace std;

const int N = 5005, M = 100005, INF = 0x3f3f3f3f;

struct Edge {
    int to, next, cap, flow, cost;
} edge[M];

int head[N], pre[N], edge_idx[N], dist[N];
bool vis[N];
int n, m, s, t, tot = 1;

void add_edge(int u, int v, int c, int w) {
    edge[++tot] = {v, head[u], c, 0, w}; head[u] = tot;
    edge[++tot] = {u, head[v], 0, 0, -w}; head[v] = tot;
}

bool spfa() {
    memset(dist, 0x3f, sizeof dist);
    memset(vis, 0, sizeof vis);
    queue<int> q;
    q.push(s); dist[s] = 0; vis[s] = 1;

    while (!q.empty()) {
        int u = q.front(); q.pop(); vis[u] = 0;
        for (int i = head[u]; i; i = edge[i].next) {
            int v = edge[i].to;
            if (edge[i].cap > edge[i].flow && dist[v] > dist[u] + edge[i].cost) {
                dist[v] = dist[u] + edge[i].cost;
                pre[v] = u;        // è®°å½•å‰é©±èŠ‚ç‚¹
                edge_idx[v] = i;    // è®°å½•æ˜¯é€šè¿‡å“ªæ¡è¾¹è¿‡æ¥çš„
                if (!vis[v]) { q.push(v); vis[v] = 1; }
            }
        }
    }
    return dist[t] != INF;
}

void mcmf(int &max_flow, int &min_cost) {
    while (spfa()) {
        int flow = INF;
        // 1. æ‰¾åˆ°è¯¥è·¯å¾„ä¸Šçš„æœ€å°å‰©ä½™å®¹é‡
        for (int i = t; i != s; i = pre[i])
            flow = min(flow, edge[edge_idx[i]].cap - edge[edge_idx[i]].flow);
        
        // 2. æ›´æ–°æµé‡å’Œæ€»è´¹ç”¨
        max_flow += flow;
        for (int i = t; i != s; i = pre[i]) {
            edge[edge_idx[i]].flow += flow;
            edge[edge_idx[i] ^ 1].flow -= flow;
            min_cost += flow * edge[edge_idx[i]].cost;
        }
    }
}

int main() {
    cin >> n >> m >> s >> t;
    while (m--) {
        int u, v, c, w;
        cin >> u >> v >> c >> w;
        add_edge(u, v, c, w);
    }
    int flow = 0, cost = 0;
    mcmf(flow, cost);
    cout << flow << " " << cost << endl;
    return 0;
}
